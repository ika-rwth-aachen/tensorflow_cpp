53tf::SavedModelBundleLite loadSavedModel(
54 const std::string& dir, const bool allow_growth = true,
55 const double per_process_gpu_memory_fraction = 0,
56 const std::string& visible_device_list = "") {
57
58 tf::SavedModelBundleLite saved_model;
59 tf::SessionOptions session_options = makeSessionOptions(
60 allow_growth, per_process_gpu_memory_fraction, visible_device_list);
61 tf::Status status =
62 tf::LoadSavedModel(session_options, tf::RunOptions(), dir,
63 {tf::kSavedModelTagServe}, &saved_model);
64 if (!status.ok())
65 throw std::runtime_error("Failed to load SavedModel: " + status.ToString());
66
67 return saved_model;
68}

120std::string getSavedModelNodeByLayerName(
121 const tf::SavedModelBundleLite& saved_model, const std::string& layer_name,
122 const std::string& signature = "serving_default") {
123
124 std::string node_name;
125 const tf::SignatureDef& model_def = saved_model.GetSignatures().at(signature);
126 auto inputs = model_def.inputs();
127 auto outputs = model_def.outputs();
128 auto& nodes = inputs;
129 nodes.insert(outputs.begin(), outputs.end());
130 for (const auto& node : nodes) {
131 const std::string& key = node.first;
132 const tf::TensorInfo& info = node.second;
133 if (key == layer_name) {
134 node_name = info.name();
135 break;
136 }
137 }
138
139 return node_name;
140}

194std::vector<std::string> getSavedModelInputNames(
195 const tf::SavedModelBundleLite& saved_model, const bool layer_names = false,
196 const std::string& signature = "serving_default") {
197
198 std::vector<std::string> names;
199 const tf::SignatureDef& model_def = saved_model.GetSignatures().at(signature);
200 for (const auto& node : model_def.inputs()) {
201 const std::string& key = node.first;
202 const tf::TensorInfo& info = node.second;
203 names.push_back(info.name());
204 }
205 std::sort(names.begin(), names.end());
206
207 if (layer_names) {
208 std::vector<std::string> node_names = names;
209 names = {};
210 for (const auto& node_name : node_names)
211 names.push_back(
212 getSavedModelLayerByNodeName(saved_model, node_name, signature));
213 }
214
215 return names;
216}

269std::vector<int> getSavedModelNodeShape(
270 const tf::SavedModelBundleLite& saved_model, const std::string& node_name,
271 const std::string& signature = "serving_default") {
272
273 std::vector<int> node_shape;
274 const tf::SignatureDef& model_def = saved_model.GetSignatures().at(signature);
275 auto inputs = model_def.inputs();
276 auto outputs = model_def.outputs();
277 auto& nodes = inputs;
278 nodes.insert(outputs.begin(), outputs.end());
279 for (const auto& node : nodes) {
280 const std::string& key = node.first;
281 const tf::TensorInfo& info = node.second;
282 if (info.name() == node_name) {
283 const auto& shape = info.tensor_shape();
284 for (int d = 0; d < shape.dim_size(); d++)
285 node_shape.push_back(shape.dim(d).size());
286 break;
287 }
288 }
289
290 return node_shape;
291}

335std::string getSavedModelInfoString(
336 const tf::SavedModelBundleLite& saved_model) {
337
338 std::stringstream ss;
339 ss << "SavedModel Info:" << std::endl;
340
341 ss << "Signatures:" << std::endl;
342 const auto& signatures = saved_model.GetSignatures();
343 for (const auto& sig : signatures) {
344
345 ss << " " << sig.first << std::endl;
346 const auto& def = sig.second;
347
348 ss << " Inputs: " << def.inputs_size() << std::endl;
349 for (const auto& node : def.inputs()) {
350 ss << " " << node.first << ": " << node.second.name() << std::endl;
351 ss << " Shape: [ ";
352 for (int d = 0; d < node.second.tensor_shape().dim_size(); d++) {
353 ss << node.second.tensor_shape().dim(d).size() << ", ";
354 }
355 ss << "]" << std::endl;
356 ss << " DataType: " << tf::DataTypeString(node.second.dtype())
357 << std::endl;
358 }
359
360 ss << " Outputs: " << def.outputs_size() << std::endl;
361 for (const auto& node : def.outputs()) {
362 ss << " " << node.first << ": " << node.second.name() << std::endl;
363 ss << " Shape: [ ";
364 for (int d = 0; d < node.second.tensor_shape().dim_size(); d++) {
365 ss << node.second.tensor_shape().dim(d).size() << ", ";
366 }
367 ss << "]" << std::endl;
368 ss << " DataType: " << tf::DataTypeString(node.second.dtype())
369 << std::endl;
370 }
371 }
372
373 return ss.str();
374}
