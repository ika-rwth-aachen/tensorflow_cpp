86 void loadModel(const std::string& filepath, const bool allow_growth = true,
87 const double per_process_gpu_memory_fraction = 0,
88 const std::string& visible_device_list = "") {
89
90 is_frozen_graph_ = (filepath.substr(filepath.size() - 3) == ".pb");
91 is_saved_model_ = !is_frozen_graph_;
92
93 // load model
94 if (is_frozen_graph_) {
95 graph_def_ = loadFrozenGraph(filepath);
96 session_ = createSession(allow_growth, per_process_gpu_memory_fraction,
97 visible_device_list);
98 loadGraphIntoSession(session_, graph_def_);
99 } else {
100 saved_model_ =
101 loadSavedModel(filepath, allow_growth, per_process_gpu_memory_fraction,
102 visible_device_list);
103 session_ = saved_model_.GetSession();
104 }
105
106 // automatically find inputs and outputs
107 if (is_frozen_graph_) {
108 input_names_ = getGraphInputNames(graph_def_);
109 output_names_ = getGraphOutputNames(graph_def_);
110 } else {
111 input_names_ = getSavedModelInputNames(saved_model_, true);
112 output_names_ = getSavedModelOutputNames(saved_model_, true);
113 const auto input_nodes_ = getSavedModelInputNames(saved_model_, false);
114 const auto output_nodes_ = getSavedModelOutputNames(saved_model_, false);
115 for (int k = 0; k < input_names_.size(); k++) {
116 saved_model_node2layer_[input_nodes_[k]] = input_names_[k];
117 saved_model_layer2node_[input_names_[k]] = input_nodes_[k];
118 }
119 for (int k = 0; k < output_names_.size(); k++) {
120 saved_model_node2layer_[output_nodes_[k]] = output_names_[k];
121 saved_model_layer2node_[output_names_[k]] = output_nodes_[k];
122 }
123 }
124 n_inputs_ = input_names_.size();
125 n_outputs_ = output_names_.size();
126
127 // run dummy inference to warm-up
128 dummyCall();
129 }

157 std::unordered_map<std::string, tf::Tensor> operator()(
158 const std::vector<std::pair<std::string, tf::Tensor>>& inputs,
159 const std::vector<std::string>& output_names) {
160
161 // properly set input/output names for session->Run()
162 std::vector<std::pair<std::string, tf::Tensor>> input_nodes;
163 std::vector<std::string> output_node_names;
164 if (is_saved_model_) {
165 for (const auto& input : inputs)
166 input_nodes.push_back(
167 {saved_model_layer2node_[input.first], input.second});
168 for (const auto& name : output_names)
169 output_node_names.push_back(saved_model_layer2node_[name]);
170 } else if (is_frozen_graph_) {
171 input_nodes = inputs;
172 output_node_names = output_names;
173 } else {
174 return {};
175 }
176
177 // run model
178 tf::Status status;
179 std::vector<tf::Tensor> output_tensors;
180 status = session_->Run(input_nodes, output_node_names, {}, &output_tensors);
181
182 // build outputs
183 std::unordered_map<std::string, tf::Tensor> outputs;
184 if (status.ok()) {
185 for (int k = 0; k < output_tensors.size(); k++)
186 outputs[output_names[k]] = output_tensors[k];
187 } else {
188 throw std::runtime_error("Failed to run model: " + status.ToString());
189 }
190
191 return outputs;
192 }

204 tf::Tensor operator()(const tf::Tensor& input_tensor) {
205
206 if (n_inputs_ != 1 || n_outputs_ != 1) {
207 throw std::runtime_error(
208 "'tf::Tensor tensorflow_cpp::Model::operator()(const tf::Tensor&)' is "
209 "only available for single-input/single-output models. Found " +
210 std::to_string(n_inputs_) + " inputs and " +
211 std::to_string(n_outputs_) + " outputs.");
212 }
213
214 // run model
215 auto outputs =
216 (*this)({{input_names_[0], input_tensor}}, {output_names_[0]});
217
218 return outputs[output_names_[0]];
219 }

231 std::vector<tf::Tensor> operator()(
232 const std::vector<tf::Tensor>& input_tensors) {
233
234 if (input_tensors.size() != n_inputs_) {
235 throw std::runtime_error(
236 "Model has " + std::to_string(n_inputs_) + " inputs, but " +
237 std::to_string(input_tensors.size()) + " input tensors were given");
238 }
239
240 // assign inputs in default order
241 std::vector<std::pair<std::string, tf::Tensor>> inputs;
242 for (int k = 0; k < n_inputs_; k++)
243 inputs.push_back({input_names_[k], input_tensors[k]});
244
245 // run model
246 auto outputs = (*this)(inputs, output_names_);
247
248 // return output tensors in default order
249 std::vector<tf::Tensor> output_tensors;
250 for (const auto& name : output_names_)
251 output_tensors.push_back(outputs[name]);
252
253 return output_tensors;
254 }

531 void dummyCall() {
532
533 // infer input shapes/types to create dummy input tensors
534 auto input_shapes = getInputShapes();
535 auto input_types = getInputTypes();
536 std::vector<tf::Tensor> input_dummies;
537 for (int k = 0; k < n_inputs_; k++) {
538 std::vector<long int> dummy_shape(input_shapes[k].begin(),
539 input_shapes[k].end());
540 // Replace -1 (batch size dimension, None in python) with 1
541 std::replace(dummy_shape.begin(), dummy_shape.end(), -1l, 1l);
542 auto dummy_tensor_shape =
543 tf::TensorShape(tf::gtl::ArraySlice<long int>(dummy_shape));
544 tf::Tensor dummy(input_types[k], dummy_tensor_shape);
545 input_dummies.push_back(dummy);
546 }
547
548 // run dummy inference
549 volatile auto output_dummies = (*this)(input_dummies);
550 }
