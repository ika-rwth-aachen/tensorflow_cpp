79 void loadModel(const std::string& filepath, const bool allow_growth = true,
80 const double per_process_gpu_memory_fraction = 0,
81 const std::string& visible_device_list = "") {
82
83 is_frozen_graph_ = (filepath.substr(filepath.size() - 3) == ".pb");
84 is_saved_model_ = !is_frozen_graph_;
85
86 // load model
87 if (is_frozen_graph_) {
88 graph_def_ = loadFrozenGraph(filepath);
89 session_ = createSession(allow_growth, per_process_gpu_memory_fraction,
90 visible_device_list);
91 loadGraphIntoSession(session_, graph_def_);
92 } else {
93 saved_model_ =
94 loadSavedModel(filepath, allow_growth, per_process_gpu_memory_fraction,
95 visible_device_list);
96 session_ = saved_model_.GetSession();
97 }
98
99 // automatically find inputs and outputs
100 if (is_frozen_graph_) {
101 input_names_ = getGraphInputNames(graph_def_);
102 output_names_ = getGraphOutputNames(graph_def_);
103 } else {
104 input_names_ = getSavedModelInputNames(saved_model_, true);
105 output_names_ = getSavedModelOutputNames(saved_model_, true);
106 const auto input_nodes_ = getSavedModelInputNames(saved_model_, false);
107 const auto output_nodes_ = getSavedModelOutputNames(saved_model_, false);
108 for (int k = 0; k < input_names_.size(); k++) {
109 saved_model_node2layer_[input_nodes_[k]] = input_names_[k];
110 saved_model_layer2node_[input_names_[k]] = input_nodes_[k];
111 }
112 for (int k = 0; k < output_names_.size(); k++) {
113 saved_model_node2layer_[output_nodes_[k]] = output_names_[k];
114 saved_model_layer2node_[output_names_[k]] = output_nodes_[k];
115 }
116 }
117 n_inputs_ = input_names_.size();
118 n_outputs_ = output_names_.size();
119
120 // run dummy inference to warm-up
121 dummyCall();
122 }

150 std::unordered_map<std::string, tf::Tensor> operator()(
151 const std::vector<std::pair<std::string, tf::Tensor>>& inputs,
152 const std::vector<std::string>& output_names) {
153
154 // properly set input/output names for session->Run()
155 std::vector<std::pair<std::string, tf::Tensor>> input_nodes;
156 std::vector<std::string> output_node_names;
157 if (is_saved_model_) {
158 for (const auto& input : inputs)
159 input_nodes.push_back(
160 {saved_model_layer2node_[input.first], input.second});
161 for (const auto& name : output_names)
162 output_node_names.push_back(saved_model_layer2node_[name]);
163 } else if (is_frozen_graph_) {
164 input_nodes = inputs;
165 output_node_names = output_names;
166 } else {
167 return {};
168 }
169
170 // run model
171 tf::Status status;
172 std::vector<tf::Tensor> output_tensors;
173 status = session_->Run(input_nodes, output_node_names, {}, &output_tensors);
174
175 // build outputs
176 std::unordered_map<std::string, tf::Tensor> outputs;
177 if (status.ok()) {
178 for (int k = 0; k < output_tensors.size(); k++)
179 outputs[output_names[k]] = output_tensors[k];
180 } else {
181 throw std::runtime_error("Failed to run model: " + status.ToString());
182 }
183
184 return outputs;
185 }

197 tf::Tensor operator()(const tf::Tensor& input_tensor) {
198
199 if (n_inputs_ != 1 || n_outputs_ != 1) {
200 throw std::runtime_error(
201 "'tf::Tensor tensorflow_cpp::Model::operator()(const tf::Tensor&)' is "
202 "only available for single-input/single-output models. Found " +
203 std::to_string(n_inputs_) + " inputs and " +
204 std::to_string(n_outputs_) + " outputs.");
205 }
206
207 // run model
208 auto outputs =
209 (*this)({{input_names_[0], input_tensor}}, {output_names_[0]});
210
211 return outputs[output_names_[0]];

224 std::vector<tf::Tensor> operator()(
225 const std::vector<tf::Tensor>& input_tensors) {
226
227 if (input_tensors.size() != n_inputs_) {
228 throw std::runtime_error(
229 "Model has " + std::to_string(n_inputs_) + " inputs, but " +
230 std::to_string(input_tensors.size()) + " input tensors were given");
231 }
232
233 // assign inputs in default order
234 std::vector<std::pair<std::string, tf::Tensor>> inputs;
235 for (int k = 0; k < n_inputs_; k++)
236 inputs.push_back({input_names_[k], input_tensors[k]});
237
238 // run model
239 auto outputs = (*this)(inputs, output_names_);
240
241 // return output tensors in default order
242 std::vector<tf::Tensor> output_tensors;
243 for (const auto& name : output_names_)
244 output_tensors.push_back(outputs[name]);
245
246 return output_tensors;
247 }

524 void dummyCall() {
525
526 // infer input shapes/types to create dummy input tensors
527 auto input_shapes = getInputShapes();
528 auto input_types = getInputTypes();
529 std::vector<tf::Tensor> input_dummies;
530 for (int k = 0; k < n_inputs_; k++) {
531 std::vector<long int> dummy_shape(input_shapes[k].begin(),
532 input_shapes[k].end());
533 // Replace -1 (batch size dimension, None in python) with 1
534 std::replace(dummy_shape.begin(), dummy_shape.end(), -1l, 1l);
535 auto dummy_tensor_shape =
536 tf::TensorShape(tf::gtl::ArraySlice<long int>(dummy_shape));
537 tf::Tensor dummy(input_types[k], dummy_tensor_shape);
538 input_dummies.push_back(dummy);
539 }
540
541 // run dummy inference
542 volatile auto output_dummies = (*this)(input_dummies);
543 }
