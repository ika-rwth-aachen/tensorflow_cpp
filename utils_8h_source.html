48tf::SessionOptions makeSessionOptions(
49 const bool allow_growth = true,
50 const double per_process_gpu_memory_fraction = 0,
51 const std::string& visible_device_list = "") {
52
53 tf::SessionOptions options = tf::SessionOptions();
54 tf::ConfigProto* config = &options.config;
55 tf::GPUOptions* gpu_options = config->mutable_gpu_options();
56 gpu_options->set_allow_growth(allow_growth);
57 gpu_options->set_per_process_gpu_memory_fraction(
58 per_process_gpu_memory_fraction);
59 gpu_options->set_visible_device_list(visible_device_list);
60
61 return options;
62}

74tf::Session* createSession(const bool allow_growth = true,
75 const double per_process_gpu_memory_fraction = 0,
76 const std::string& visible_device_list = "") {
77
78 tf::Session* session;
79 tf::SessionOptions options = makeSessionOptions(
80 allow_growth, per_process_gpu_memory_fraction, visible_device_list);
81 tf::Status status = tf::NewSession(options, &session);
82 if (!status.ok())
83 throw std::runtime_error("Failed to create new session: " +
84 status.ToString());
85
86 return session;
87}
